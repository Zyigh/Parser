5! = 5 * 4 * 3 * 2 * 1
function factoriel(n) {
    resultat = 1
    Tant que n > 1 :
        resultat = resultat * n
        n = n-1

    return resultat
}

function factoriel(n) {
    SI n = 1 :
        return 1

    return n * factoriel (n-1)
